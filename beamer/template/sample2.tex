					\label{alg:maxmin}
					\DontPrintSemicolon
					\SetKwData{A}{ants}
					\SetKwData{a}{ant}
					\SetKwData{ib}{iteration-best}
					\SetKwData{gb}{global-best}
					\SetKwData{b}{best-ant}
					\SetKwData{I}{instanceSize}
					\SetKwData{PM}{pheromone-matrix}
					\SetKwFunction{C}{ConstructSolution}
					\SetKwFunction{Cost}{Cost}
					\SetKwFunction{L}{LocalSearch}
					\SetKwFunction{P}{UpdatePheromoneMatrix}
					\SetKwFunction{IPM}{InitializePheromoneMatrix}
					\SetKwFunction{SB}{SelectBest}
					\SetKwFunction{UMM}{UpdateMaxMinValues}
					$m \gets \mbox{number of ants}$\;
					$\gb \gets \infty$\;
					$\PM \gets \IPM{}$\;
					\Repeat{stopping criteria is not met}{%
						$\ib \gets \infty$\;
						\For{$k \gets 0$ \KwTo $m$}{
							$\a \gets \C{}$\;
							$\a \gets \L{\a}$\;
							\If{\Cost{\a} $<$ \Cost{\ib}}{%
								$\ib \gets \a$\;
							}
						}
						\If{\Cost{\ib} $<$ \Cost{\gb}}{%
							$\gb \gets \ib$\;
							$\UMM{}$\;
						}
						$\b \gets \SB{\gb, \ib}$\;
						$\P{\b}$\;
					}
					\Return \gb
				\end{algorithm}
